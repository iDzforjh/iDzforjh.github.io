Screeps

2020年10月25日 440 字 大概 2 分钟

Screeps-面向程序员的策略游戏

创建工人，其具有 work carry move等三种属性

Game.spawns['Spawn1'].spawnCreep( [WORK, CARRY, MOVE], 'Harvester1' );

让工人执行最基本的挖矿脚本

module.exports.loop = function () {
    var creep = Game.creeps['Harvester1'];
    var sources = creep.room.find(FIND_SOURCES);
    if(creep.harvest(sources[0]) == ERR_NOT_IN_RANGE) {
        creep.moveTo(sources[0]);
    }
}

Game.creeps[ worker_name ] 获取需要被控制的 creep

creep.room.find find 通过 room来调用

creep.harvest(sources[0]) == ERR_NOT_IN_RANGE (sources是一个数组并不是对象)

ERR_NOT_IN_RANGE 目标距离 creep 太远 （注：soucre 似乎是第一个资源点的距离（相对creep））

采矿执行完毕，返回基地

module.exports.loop = function () {
    var creep = Game.creeps['Harvester1'];

    if(creep.store.getFreeCapacity() > 0) {
        var sources = creep.room.find(FIND_SOURCES);
        if(creep.harvest(sources[0]) == ERR_NOT_IN_RANGE) {
            creep.moveTo(sources[0]);
        }
    }
    else {
        if( creep.transfer(Game.spawns['Spawn1'], RESOURCE_ENERGY) == ERR_NOT_IN_RANGE ) {
            creep.moveTo(Game.spawns['Spawn1']);
        }
    }
}

if(creep.store.getFreeCapacity() > 0) :

  • .store 代表其存储中的资源对象
  • .getFreeCapacity() 方法 返回剩余可用容量

creep.transfer(Game.spawns['Spawn1'], RESOURCE_ENERGY) == ERR_NOT_IN_RANGE

  • .transfer(target, resourcetype, [amount]（可选）) 将资源从该creep转移至其他对象，目标必须在紧邻creep的正方形区域中

遍历所有的 creeps 执行挖矿代码

module.exports.loop = function () {
    for(var name in Game.creeps) {
        var creep = Game.creeps[name];

        if(creep.store.getFreeCapacity() > 0) {
            var sources = creep.room.find(FIND_SOURCES);
            if(creep.harvest(sources[0]) == ERR_NOT_IN_RANGE) {
                creep.moveTo(sources[0]);
            }
        }
        else {
            if(creep.transfer(Game.spawns['Spawn1'], RESOURCE_ENERGY) == ERR_NOT_IN_RANGE) {
                creep.moveTo(Game.spawns['Spawn1']);
            }
        }
    }
}

模块

左下角新建模块（函数）

固定模版格式 run：function(){} 

var roleHarvester = require('role.harvester');

module.exports.loop = function () {

    for(var name in Game.creeps) {
        var creep = Game.creeps[name];
        roleHarvester.run(creep);
    }
}

var roleHarvester = require('role.harvester'); 模块引用

roleHarvester.run(creep); 调用 模块 里的函数方法
